                <div className="mt-8 p-6 bg-slate-50 rounded-2xl border border-slate-200">
                   <h4 className="font-bold text-sm text-slate-700 mb-3 flex items-center gap-2">
                     <ShieldAlert className="w-4 h-4 text-red-500" />
                     إرسال تنبيه إداري (يتطلب تأكيد استلام)
                    </h4>
                   <div className="flex flex-col gap-3">
                     <Input 
                       placeholder="اكتب رسالة رسمية للعامل... سيصل التنبيه بصوت مرتفع جداً"
                       className="bg-white border-slate-200 rounded-xl"
                       id="admin-notif-input"
                     />
                     <div className="flex gap-2">
                         <Button 
                           className="rounded-xl bg-red-600 hover:bg-red-700 font-bold flex-1"
                           onClick={async () => {
                             const input = document.getElementById('admin-notif-input') as HTMLInputElement;
                             if (!input.value.trim()) return;
                             try {
                               await addDoc(collection(db, 'notifications'), {
                                 title: 'تنبيه إداري رسمي',
                                 message: input.value.trim(),
                                 workerId,
                                 priority: 'critical',
                                 requiresAcknowledge: true,
                                 type: 'error',
                                 category: 'system',
                                 timestamp: serverTimestamp(),
                                 createdBy: profile?.uid,
                                 targetRole: 'employee'
                               });
                               toast.success('تم إرسال التنبيه الرسمي بنجاح');
                               input.value = '';
                             } catch (e) {
                               toast.error('فشل في إرسال التنبيه');
                             }
                           }}
                         >
                           إرسال تنبيه عاجل
                         </Button>
                         <Button 
                           variant="outline"
                           className="rounded-xl font-bold"
                           onClick={async () => {
                             const input = document.getElementById('admin-notif-input') as HTMLInputElement;
                             if (!input.value.trim()) return;
                             try {
                               await addDoc(collection(db, 'notifications'), {
                                 title: 'رسالة إدارية',
                                 message: input.value.trim(),
                                 workerId,
                                 priority: 'medium',
                                 requiresAcknowledge: false,
                                 type: 'info',
                                 category: 'system',
                                 timestamp: serverTimestamp(),
                                 createdBy: profile?.uid,
                                 targetRole: 'employee'
                               });
                               toast.success('تم إرسال الرسالة بنجاح');
                               input.value = '';
                             } catch (e) {
                               toast.error('فشل في إرسال الرسالة');
                             }
                           }}
                         >
                           إرسال رسالة عادية
                         </Button>
                     </div>
                   </div>
                </div>
